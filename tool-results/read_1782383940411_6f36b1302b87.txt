    45→function daysUntil(date: Date | null): number | null {
    46→  if (!date) return null;
    47→  const now = new Date();
    48→  const diff = date.getTime() - now.getTime();
    49→  return Math.ceil(diff / (1000 * 60 * 60 * 24));
    50→}
    51→
    52→function formatExperienceLabel(level: string | null | undefined): string {
    53→  if (!level) return 'Not specified';
    54→  const map: Record<string, string> = {
    55→    ENTRY: 'Entry level (0 years experience)',
    56→    JUNIOR: 'Junior (1-2 years experience)',
    57→    MID: 'Mid-level (3-5 years experience)',
    58→    SENIOR: 'Senior (5-8 years experience)',
    59→    LEAD: 'Lead (8-10 years experience)',
    60→    EXECUTIVE: 'Executive (10+ years experience)',
    61→  };
    62→  return map[level] || level;
    63→}
    64→
    65→function formatEducationLabel(level: string | null | undefined): string {
    66→  if (!level) return 'Not specified';
    67→  const map: Record<string, string> = {
    68→    NONE: 'No formal education requirement',
    69→    HIGH_SCHOOL: 'Kenya Certificate of Secondary Education (KCSE)',
    70→    CERTIFICATE: 'Professional Certificate',
    71→    DIPLOMA: 'Diploma from a recognised institution',
    72→    BACHELORS: 'Bachelor\'s degree',
    73→    MASTERS: 'Master\'s degree',
    74→    DOCTORATE: 'Doctorate (PhD)',
    75→    PROFESSIONAL: 'Professional qualification or certification',
    76→  };
    77→  return map[level] || level;
    78→}

    80→function deriveResponsibilities(job: FormattedJob): string[] {
    81→  if (job.responsibilities && job.responsibilities.length > 0) return job.responsibilities;
    82→  const base = [
    83→    `Execute core ${job.title} duties and deliver on key performance indicators set by ${job.company}`,
    84→    `Collaborate with cross-functional teams across departments to drive ${job.category ? job.category.toLowerCase() : 'business'} objectives`,
    85→    `Prepare regular reports, documentation, and presentations for management review and stakeholder communication`,
    86→  ];
    87→  if (job.experienceLevel === 'SENIOR' || job.experienceLevel === 'LEAD' || job.experienceLevel === 'EXECUTIVE') {
    88→    base.push(
    89→      `Provide strategic leadership and mentorship to junior team members within the ${job.category || 'department'}`,
    90→      `Drive process improvements and best practices to enhance operational efficiency`
    91→    );
    92→  }
    93→  if (job.isRemote) {
    94→    base.push('Participate effectively in remote team meetings and maintain clear communication across distributed teams');
    95→  }
    96→  return base;
    97→}
    98→
    99→function deriveRequirements(job: FormattedJob): string[] {
   100→  if (job.requirements && job.requirements.length > 0) return job.requirements;
   101→  const base: string[] = [];
   102→  if (job.educationLevel && job.educationLevel !== 'NONE') {
   103→    base.push(`${formatEducationLabel(job.educationLevel)} in a relevant field${job.category ? ' such as ' + job.category.toLowerCase() : ''}`);
   104→  } else {
   105→    base.push(`Minimum of a Diploma or equivalent qualification in a relevant field`);
   106→  }
   107→  base.push(formatExperienceLabel(job.experienceLevel));
   108→  if (job.category && (job.category.includes('Technology') || job.category.includes('IT') || job.category.includes('Software'))) {
   109→    base.push('Proficiency in relevant programming languages, tools, and technologies');
   110→    base.push('Strong problem-solving and analytical thinking abilities');
   111→  }
   112→  base.push('Excellent written and verbal communication skills in English and Kiswahili');
   113→  base.push('Ability to work independently and collaboratively in a fast-paced environment');
   114→  if (job.isRemote) {
   115→    base.push('Reliable high-speed internet connection and a dedicated workspace for remote work');
   116→  }
   117→  if (job.locationCounty) {
   118→    base.push(`Based in or willing to relocate to ${job.locationCounty}${job.isRemote ? ' (remote work available)' : ''}`);
   119→  }
   120→  return base;
   121→}
