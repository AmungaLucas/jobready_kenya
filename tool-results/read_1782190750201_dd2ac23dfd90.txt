  1488→// services/job.service.ts
  1489→export async function createJob(data: CreateJobInput) {
  1490→  const org = await prisma.organization.findUnique({
  1491→    where: { id: data.organizationId },
  1492→    select: { orgName: true },
  1493→  });
  1494→
  1495→  const subcategory = await prisma.jobSubcategory.findUnique({
  1496→    where: { id: data.subcategoryId },
  1497→    select: { categoryId: true },
  1498→  });
  1499→
  1500→  const searchText = [
  1501→    data.title,
  1502→    data.description,
  1503→    org?.orgName || '',
  1504→    data.locationCity,
  1505→    data.locationCounty,
  1506→  ].filter(Boolean).join(' ');
  1507→
  1508→  return prisma.job.create({
  1509→    data: {
  1510→      ...data,
  1511→      categoryId: subcategory?.categoryId,
  1512→      searchText,
  1513→    },
  1514→  });
  1515→}

  1518→### ✅ MySQL Full-Text Search
  1519→```ts
  1520→export async function searchJobs(query: string) {
  1521→  return prisma.$queryRaw`
  1522→    SELECT j.*, o."orgName", o."orgLogoUrl"
  1523→    FROM "jobs" j
  1524→    LEFT JOIN "organizations" o ON j."organizationId" = o.id
  1525→    WHERE 
  1526→      j."status" = 'ACTIVE' 
  1527→      AND j."deletedAt" IS NULL
  1528→      AND MATCH(j."search_text") AGAINST(${query} IN NATURAL LANGUAGE MODE)
  1529→    ORDER BY MATCH(j."search_text") AGAINST(${query} IN NATURAL LANGUAGE MODE) DESC
  1530→    LIMIT 50
  1531→  `;
  1532→}
  1533→```
